    Quick Start¶

    This guide walks you through building a .NET application with Port step by step.

    Requirements¶

    Requirement Specification
    OS Windows 11+ / Windows Server 2003+
    Memory Minimum 32GB
    .NET SDK .NET 8.0+
    Port v1.1.78 Download
    NuGet portdic package

    Step 1. Create a New Project¶

    dotnet new wpf -n MyEquipment
cd MyEquipment
dotnet add package portdic

    dotnet new winforms -n MyEquipment
cd MyEquipment
dotnet add package portdic

    dotnet new console -n MyEquipment
cd MyEquipment
dotnet add package portdic

    Step 2. New Repository with Port CLI¶

    port new myEQ

    Creating Pages from Documents¶

    Instead of writing .page files manually, you can auto-generate them from Excel/Word documents using the [Document] attribute. The wapis project uses this approach to generate device/io.page from IO_Map.docx.

    // Model: maps Excel columns to entry fields
public class IOModel
{
    [ColumnHeader, EntryProperty]
    public string Pin_No { set; get; } = string.Empty;

    [ColumnHeader, EntryProperty]
    public string Port_NO { set; get; } = string.Empty;

    [ColumnHeader, EntryKey]
    public string Description { set; get; } = string.Empty;

    [ColumnHeader, EntryProperty]
    public string Model { set; get; } = string.Empty;

    [ColumnHeader, EntryProperty]
    public string Bit_On { set; get; } = string.Empty;

    [EntryDataType]
    public string DataType { set; get; } = "enum.OffOn";
}

// Converter: reads docx, outputs .page + .cs
[Document(@"D:\PORT\SampleArduinoLib\wapis\port\IO_Map.docx")]
public class IODocument
{
    [Save(@"D:\PORT\SampleArduinoLib\wapis\port\device\io.page", @"D:\PORT\SampleArduinoLib\wapis\port\io1.cs")]
    public Document<IOModel> Convert(Document<IOModel> doc)
    {
        doc.ForEach(v => v.DataType = "enum.OffOn");
        doc.ForEach(v => v.Package = "DigitalIO.DI");
        return doc;
    }
}

Door_Lock_On_Check              enum.OffOn  pkg:DigitalIO.DI  property:{"Pin_No":"6","Port_NO":"X01.05","Model":"G7SA-2A2B","Bit_On":"Lock"}
FFU1_Normal_Status              enum.OffOn  pkg:DigitalIO.DI  property:{"Pin_No":"15","Port_NO":"X01.14","Model":"MS-FC300","Bit_On":"Normal"}

    Step 3. Push to Repository¶

    After writing your data pages, push them to the database.

    port push

    This command parses .page files and registers them in Port's in-memory database.

    Step 4. Initialize Port in Your Application¶

    Example A — WPF Application (wapis pattern)¶

    using portdic;
using System.Windows;

namespace MyEquipment
{
    public partial class MainWindow : Window
    {
        public MainWindow()
        {
            InitializeComponent();

            // 1. Load repository
            Port.Repository.Load("myequipment");

            // 2. Register packages
            Port.Add<Bulb>("Bulb1");
            Port.Add<Heater>("Heater1");

            // 3. Subscribe to ready event
            Port.OnReady += Port_OnReady;

            // 4. Start the server
            Port.Run();
        }

        private void Port_OnReady(object? sender, EventArgs e)
        {
            // Port server synchronized — Set/Get operations are now available
        }
    }
}

    Example B — WinForms Application (DemoDotNetApp pattern)¶

    using portdic;
using System.Windows.Forms;

namespace MyEquipment
{
    public partial class FormMain : Form
    {
        public FormMain()
        {
            InitializeComponent();

            // 1. Register packages
            Port.Add<Bulb>("Bulb1");
            Port.Add<Heater>("Heater1");

            // 2. Subscribe to events
            Port.OnReady += Port_OnReady;

            // 3. Start the server
            Port.Run();
        }

        private void Port_OnReady(object sender, EventArgs e)
        {
            // Port server synchronized
        }
    }
}

    Initialization Order

    You must follow the order: Port.Add<T>() → Port.OnReady += → Port.Run(). Once Port.Run() is called, the server starts. The OnReady event fires when synchronization is complete.

    Step 5. Create a Package¶

    Use the [Package] attribute to define equipment libraries. Port automatically manages instances and generates APIs.

    Basic Package¶

    using portdic;
using portdatatype;

[Package]
public class Bulb
{
    [Logger]
    public ILogger Logger { get; set; }

    [Property]
    public IProperty Property { get; set; }

    [Valid("Device not connected")]
    public bool Valid() => true;

    private string _status = "Off";

    [API(EntryDataType.Enum)]
    public string OffOn
    {
        get => _status;
        set
        {
            _status = value;
            Logger.Write($"Bulb status changed to {value}");
        }
    }
}

    Temperature Sensor Package¶

    using portdic;
using portdatatype;

[Package]
public class Heater
{
    [Logger]
    public ILogger Logger { get; set; }

    [Property]
    public IProperty Property { get; set; }

    [Valid("Invalid for connection")]
    public bool Valid() => true;

    [API(EntryDataType.Num, PropertyFormat.Json, "Unit")]
    public double Temp1
    {
        get
        {
            if (Property?.TryToGetValue("Unit", out string unit) == true)
            {
                // Return temperature based on unit
                return unit == "F" ? 72.5 : 22.5;
            }
            return double.NaN;
        }
    }
}
    Attribute Purpose
    [Package] Registers the class as a Port-managed package
    [Logger] Injects ILogger automatically
    [Property] Injects IProperty for configuration access
    [Valid("msg")] Defines package validation logic
    [API(type)] Generates REST API endpoint automatically

    See the full attribute list in the Attribute Reference.

    Step 6. Set / Get Values¶

    After the Port server reaches the OnReady state, you can read and write data.

    Set — Write Values¶

    // Method 1: category, key, value
Port.Set("room1", "BulbOnOff", "On");

// Method 2: dot notation
Port.Set("room1.BulbOnOff", "On");

// Numeric values
Port.Set("room1", "RoomTemp1", 42.5);

    Get — Read Values¶

    // Method 1: category, key
var value = Port.Get("room1", "BulbOnOff");
string status = value.String();     // "On"

// Method 2: dot notation
var temp = Port.Get("room1.RoomTemp1");
string text = temp.String();        // "42.5"

    Using Token Constants¶

    Use constant classes for type-safe key references.

    // tokens.cs
namespace portdic
{
    public static class Room1
    {
        public const string BulbOnOff = "room1.BulbOnOff";
        public const string RoomTemp1 = "room1.RoomTemp1";
        public const string RoomTemp2 = "room1.RoomTemp2";
    }
}

    // Usage
Port.Set(Room1.BulbOnOff, "On");
var temp = Port.Get(Room1.RoomTemp1).String();

    Step 7. Run Your Application¶

    Run via CLI¶

    port run myequipment

    Run as GUI Application¶

    Launch your project with Visual Studio or dotnet run — the Port server starts automatically.

    dotnet run --project MyEquipment

    Error Inspection¶

    port run myequipment --ng ignore

    Detailed error analysis: http://localhost:5001/api/app/ng/?view=table

    Advanced Examples¶

    Flow Control¶

    Define step-by-step automation flows like the LoadPortController in DemoDotNetApp.

    using portdic;

[FlowController]
public partial class LoadPortController
{
    [Flow]
    public class LoadFlow
    {
        [FlowModel]
        public IFlowModel model { get; set; }

        [FlowNotify]
        public IFlowNotify Notify { get; set; }

        [FlowStep]
        [FlowWatcherCompare("@Temp1", ">=", 50)]
        [FlowWatcherAction("room2.BulbOnOff", "Off")]
        public void Step1()
        {
            Task.Delay(1000).Wait();
            Notify.Next();
        }

        [FlowStep(nameof(Step1))]
        [FlowWatcherCompare("room2.HeaterTemp4", ">=", 100)]
        [FlowWatcherAction("room2.BulbOnOff", "On")]
        public void Step2()
        {
            Task.Delay(1000).Wait();
            Notify.Next();
        }
    }
}

    // Execute a flow
Port.Set("LP1.LoadFlow", FlowAction.Executing);

// Check flow status
var controller = Port.GetController("LP1");
if (controller.Status == ControlStatus.Executing)
{
    controller.AbortAllFlow();
}

    SECS/GEM Communication¶

    Set up semiconductor equipment communication like the GemHelper in wapis.

    using portdic;
using portdic.secs;

[GEM]
public class GemHelper
{
    [GemHandler]
    public IGemHandler handler { get; set; } = null!;

    [Preset]
    private void Preset()
    {
        handler.SetMode(Mode.Passive);
        handler.SetAddress("127.0.0.1:6000");
        handler.SetDevice("0");
        handler.SetT3(45);

        handler.OnS10F1_TerminalDisplaySingle += OnTerminalDisplay;
    }

    private void OnTerminalDisplay(IGemReplier replier, ReceiveSecsMessage message)
    {
        // Handle terminal message from host
    }
}

    // Register in MainWindow
Port.Add<GemHelper>("GEM");

    Protocol Broadcasting¶

    // MQTT
var mqtt = Port.BroadCast<MQTT>(
    Port.CreateBuilder(BroadcastType.MQTT)
        .Address("127.0.0.1:8080")
        .Users(new User("admin", "admin"))
        .Build());

// OPC UA
var opcua = Port.BroadCast<OPCUA>(
    Port.CreateBuilder(BroadcastType.OPCUA)
        .Address("opc.tcp://localhost:4840")
        .Build());
